# baresip-villa


## Building instructions

	cd re
	cmake -B build
	cmake --build build
	sudo cmake --install build
	sudo ldconfig
	cd ..

	cd libfvad
	cmake -B build
	cmake --build build
	cd ..

	cd baresip-apps
	cmake -B build
	cmake --build build
	sudo cmake --install build
	cd ..

	cd baresip
	git checkout immisch/aufile-src-offset
	cmake -B build
	cmake --build build
	sudo cmake --install build
	cd ..
